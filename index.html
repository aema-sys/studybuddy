-- Tabellen aanmaken CREATE TABLE IF NOT EXISTS flashcards ( id BIGSERIAL PRIMARY KEY, vak TEXT NOT NULL, week TEXT NOT NULL, theme TEXT NOT NULL, question TEXT NOT NULL, answer TEXT NOT NULL, mat_type TEXT DEFAULT 'slides', practiced INT DEFAULT 0, correct INT DEFAULT 0, last_rating INT DEFAULT 0, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS quiz_sessions ( id BIGSERIAL PRIMARY KEY, total INT NOT NULL, correct INT NOT NULL, vak TEXT, mode TEXT DEFAULT 'type', created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS notes ( id BIGSERIAL PRIMARY KEY, vak TEXT, week TEXT, content TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Beveiliging ALTER TABLE flashcards ENABLE ROW LEVEL SECURITY; ALTER TABLE quiz_sessions ENABLE ROW LEVEL SECURITY; ALTER TABLE notes ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "allow_all_flashcards" ON flashcards; DROP POLICY IF EXISTS "allow_all_sessions" ON quiz_sessions; DROP POLICY IF EXISTS "allow_all_notes" ON notes; CREATE POLICY "allow_all_flashcards" ON flashcards FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all_sessions" ON quiz_sessions FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all_notes" ON notes FOR ALL USING (true) WITH CHECK (true);
